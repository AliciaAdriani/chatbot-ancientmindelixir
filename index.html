  <script>

    const chatlog = document.getElementById('chatlog');

    const responses = {

      "hi": "Hello! I'm your AncientMind assistant. How can I support your journey today?",

      "hello": "Hi there! How can I help you today?",

      "how to use": "Take 1 capsule daily at any time that works best for you. Consistency is key!",

      "refund": "We'd love to help you experience the full benefits! Many people notice deeper results after 3-4 weeks. Would you consider continuing a bit longer? If you still wish to proceed, please email us at support@ancientmindelixir.com.",

      "support": "Our team is ready to assist. Email us anytime at support@ancientmindelixir.com."

    };

    function appendMessage(text, sender = 'bot') {

      const msg = document.createElement('div');

      msg.className = 'message ' + sender;

      msg.innerText = text;

      chatlog.appendChild(msg);

      chatlog.scrollTop = chatlog.scrollHeight;

    }

    function sendMessage() {

      const input = document.getElementById('userInput');

      const text = input.value.trim();

      if (!text) return;

      appendMessage(text, 'user');

      const lower = text.toLowerCase();

      const key = Object.keys(responses).find(k => lower.includes(k));

      const reply = key ? responses[key] : "I'm not sure I understand. Please try asking another way, or contact us at support@ancientmindelixir.com.";

      setTimeout(() => appendMessage(reply, 'bot'), 500);

      input.value = '';

    }

    appendMessage("Hi there! I'm your AncientMind assistant. Ask me anything about your journey, the elixir, or support.");
